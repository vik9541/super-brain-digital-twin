        # –ù–∞—Ö–æ–¥–∏–º –æ—Ä–≥–∞–Ω–∏–∑–∞—Ü–∏–∏ —Å –±–æ–ª—å—à–∏–º —á–∏—Å–ª–æ–º –∫–æ–Ω—Ç–∞–∫—Ç–æ–≤

            'top_email_domains': sorted(email_domains.items(), key=lambda x: x[1], reverse=True)[:5],
